1. jquely

htmlでjqueryを呼び出す際はhead内のmeta,titleに続いて先頭であると他の外部からJSを呼び出した場合そのJSで jqueryを扱うならその方が望ましい。用は上から順に呼び出されるためである。

<!DOCTYPE html>
<html>
    <head>
        <meta charset="utf-8">
        <title>好きなタイトル</title>
        <script src="https://code.jquery.com/jquery-3.2.1.min.js"></script>

        <script type="text/javascript" src="読み込みたいJS.js" charset="utf-8"></script>
        <script type="text/javascript" src="読み込みたいJS.js" charset="utf-8"></script>

3. JS時刻の取り方

JSファイルにおいて。（先頭に０がつくよう整地する関数も付属）

function set2fig(num) {
    // 桁数が1桁だったら先頭に0を加えて2桁に調整する
    var ret;
    if( num < 10 ) { ret = "0" + num; }
    else { ret = num; }
    return ret;
 }


var nowTime = new Date();
var nowHour = set2fig( nowTime.getHours() );
var nowMin  = set2fig( nowTime.getMinutes() );
var nowSec  = set2fig( nowTime.getSeconds() );
var msg = "現在時刻は、" + nowHour + ":" + nowMin + ":" + nowSec + " です。";

4. JS・UNIXTIMEへの変換

getTimeの場合ミリ秒単位なので秒単位に変換する必要がある。

var date = new Date() ;

// UNIXタイムスタンプを取得する (ミリ秒単位)
var unixsec = date.getTime() ;

// UNIXタイムスタンプを取得する (秒単位)
var unixtime = Math.floor( unixsec / 1000 ) ;

9.繰り返し処理の方法と豆

JSで繰り返し秒指定ごとにとってくるには

setInterval(function(){
    //何かしらの処理
},1000);

1000で1秒なのでこれに掛け算して指定秒数になるようにする。
